# Teensy MIDI to HID Keyboard Translator Configuration File
# Place this file on the SD card in the Teensy 4.1
# File name must be: CONFIG.TXT
#
# If this file is missing, default values will be used:
#   FAST_PRESS_MODE=true
#   PRESS_DURATION=0
#   PROFILE_SWITCH_NOTE=24 (C1)

# Fast-press mode: If true, keys are pressed/released quickly regardless of MIDI duration
# Useful for games that don't recognize held keys (like Where Winds Meet)
# Values: true, false, 1, 0, ON, OFF, YES, NO (case-insensitive)
FAST_PRESS_MODE=true

# Press duration: How long to hold keys in fast-press mode (milliseconds)
# 0 = immediate press/release (recommended for most games)
# 1-1000 = hold for specified duration before releasing
# Only applies when FAST_PRESS_MODE=true
PRESS_DURATION=0

# Profile switch note: MIDI note number to trigger profile switching
# Default: 24 (C1) - change to match your keyboard's lowest C key
# Valid range: 0-127 (any MIDI note)
# Press this MIDI note to cycle through available profiles in mapping files
# Set to 255 to disable profile switching
PROFILE_SWITCH_NOTE=24

# Examples:
#
# Immediate press/release (recommended):
# FAST_PRESS_MODE=true
# PRESS_DURATION=0
#
# Hold keys for 50ms:
# FAST_PRESS_MODE=true
# PRESS_DURATION=50
#
# Normal mode (hold keys as long as MIDI note is held):
# FAST_PRESS_MODE=false
# PRESS_DURATION=0
